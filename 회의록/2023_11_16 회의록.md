# **Database schema로부터 Table 생성**

```sql
-- Users(사용자)
CREATE TABLE Users (
    UserID SERIAL PRIMARY KEY,
    UserName VARCHAR(20),
    UserRole VARCHAR(30),
    StudentID INTEGER,
    FOREIGN KEY (StudentID) REFERENCES Student(StudentID)
);

-- Student(원아) 
CREATE TABLE Student (
    StudentID SERIAL PRIMARY KEY,
    StudentName VARCHAR(20),
    ClassName CHAR,
    BirthDate DATE,
    Attendance INTEGER,
    HealthStatus BOOLEAN,
    Address VARCHAR(100)
);

-- Chat(알림장)
CREATE TABLE Chat (
    ChatID SERIAL PRIMARY KEY,
    SenderID INTEGER,
    ReceiverID INTEGER,
    Message TEXT,
    TimeStamp TIMESTAMP,
    Image BYTEA
);

-- Schedule(일정)
CREATE TABLE Schedule (
    ScheduleID SERIAL PRIMARY KEY,
    EventType VARCHAR(200),
    Date DATE,
    Time TIME,
    Description TEXT,
    ScheduleImg BYTEA
);

-- ScheduleStudent(스케줄-원아 매핑 테이블)
CREATE TABLE ScheduleStudent (
    ScheduleID INTEGER,
    StudentID INTEGER,
    PRIMARY KEY (ScheduleID, StudentID),
    FOREIGN KEY (ScheduleID) REFERENCES Schedule(ScheduleID),
    FOREIGN KEY (StudentID) REFERENCES Student(StudentID)
);

-- GuardianSelection(안심하원서비스)
CREATE TABLE GuardianSelection (
    SelectionID SERIAL PRIMARY KEY,
    TeacherID INTEGER,
    GuardianID INTEGER,
    StudentID INTEGER,
    FOREIGN KEY (TeacherID) REFERENCES Users(UserID),
    FOREIGN KEY (GuardianID) REFERENCES Users(UserID),
    FOREIGN KEY (StudentID) REFERENCES Student(StudentID)
);

-- FreeBoardQA(자유 게시판)
CREATE TABLE FreeBoardQA (
    PostID SERIAL PRIMARY KEY,
    PosterID INTEGER,
    Title VARCHAR(100),
    Content TEXT,
    TimeStamp TIMESTAMP,
    Image BYTEA,
    FOREIGN KEY (PosterID) REFERENCES Users(UserID)
);

-- Comment(댓글)
CREATE TABLE Comment (
    CommentID SERIAL PRIMARY KEY,
    PostID INTEGER,
    CommentContent TEXT,
    TimeStamp TIMESTAMP,
    FOREIGN KEY (PostID) REFERENCES FreeBoardQA(PostID)
);

-- MealPlan(식단표 조회)
CREATE TABLE MealPlan (
    PlanID SERIAL PRIMARY KEY,
    Date DATE,
    Meal1 TEXT,
    Meal2 TEXT,
    Snack TEXT
);
```
